66  void Split(easy::Sprite s, Si32 border_size,
67  bool is_x_scaleable, bool is_y_scaleable) {
68  // | w |
69  // 0 x1 x2 x3
70  // a---------b y3
71  // | c---d | y2 -
72  // | | | | h
73  // | e---f | y1 -
74  // g---------k 0
75  Si32 x1 = border_size;
76  Si32 x2 = s.Width() - border_size;
77  Si32 y1 = border_size;
78  Si32 y2 = s.Height() - border_size;
79  Si32 w = s.Width() - border_size * 2;
80  Si32 h = s.Height() - border_size * 2;
81  left_.Reference(s, 0, y1, border_size, h);
82  right_.Reference(s, x2, y1, border_size, h);
83  top_.Reference(s, x1, y2, w, border_size);
84  bottom_.Reference(s, x1, 0, w, border_size);
85  upper_left_.Reference(s, 0, y2, border_size, border_size);
86  upper_right_.Reference(s, x2, y2, border_size, border_size);
87  lower_left_.Reference(s, 0, 0, border_size, border_size);
88  lower_right_.Reference(s, x2, 0, border_size, border_size);
89  background_tile_.Reference(s, x1, y1, w, h);
90  is_x_scaleable_ = is_x_scaleable;
91  is_y_scaleable_ = is_y_scaleable;
92  }

94  Vec2Si32 EstimateSizeForClienArea(Vec2Si32 client_area_size) {
95  Vec2Si32 size_bound = client_area_size;
96  if (!is_x_scaleable_) {
97  size_bound.x = (client_area_size.x + top_.Width() - 1) /
98  top_.Width() * top_.Width();
99  }
100  if (!is_y_scaleable_) {
101  size_bound.y = (client_area_size.y + left_.Height() - 1) /
102  left_.Height() * left_.Height();
103  }
104  Vec2Si32 size;
105  size.x = size_bound.x + left_.Width() + right_.Width();
106  size.y = size_bound.y + top_.Height() + bottom_.Height();
107  return size;
108  }
109 
110  Vec2Si32 ClientAreaForSize(Vec2Si32 size) {
111  Vec2Si32 size_bound;
112  size_bound.x = size.x - left_.Width() - right_.Width();
113  size_bound.y = size.y - top_.Height() - bottom_.Height();
114  return size_bound;
115  }

117  easy::Sprite DrawClientSize(Vec2Si32 client_area_size) {

128  easy::Sprite DrawBothSizes(Vec2Si32 client_size, Vec2Si32 size) {
129  easy::Sprite sprite;
130  sprite.Create(size.x, size.y);
131  sprite.Clear(Rgba(0xff000000));
132  // Tile background.
133  Vec2Si32 pos(0, 0);
134  Vec2Si32 offset(left_.Width(), bottom_.Height());
135  if (background_tile_.Size().x && background_tile_.Size().y) {
136  while (pos.y < client_size.y) {
137  while (pos.x < client_size.x) {
138  Vec2Si32 instance_size(
139  std::min(background_tile_.Width(), client_size.x - pos.x),
140  std::min(background_tile_.Height(), client_size.y - pos.y));
141  background_tile_.Draw(pos.x + offset.x, pos.y + offset.y,
142  instance_size.x, instance_size.y,
143  0, 0, instance_size.x, instance_size.y, sprite, easy::kCopyRgba);
144  pos.x += background_tile_.Width();
145  }
146  pos.y += background_tile_.Height();
147  pos.x = 0;
148  }
149  }
150  for (Si32 x = 0; x < client_size.x; x += top_.Width()) {
151  Si32 width = is_x_scaleable_ ?
152  std::min(top_.Width(), client_size.x - x) :
153  top_.Width();
154  top_.Draw(x + left_.Width(), size.y - top_.Height(),
155  width, top_.Height(),
156  0, 0, width, top_.Height(), sprite, easy::kCopyRgba);
157  }
158  for (Si32 x = 0; x < client_size.x; x += bottom_.Width()) {
159  Si32 width = is_x_scaleable_ ?
160  std::min(bottom_.Width(), client_size.x - x) :
161  bottom_.Width();
162  bottom_.Draw(x + left_.Width(), 0,
163  width, bottom_.Height(),
164  0, 0, width, bottom_.Height(), sprite, easy::kCopyRgba);
165  }
166  for (Si32 y = 0; y < client_size.y; y += left_.Height()) {
167  left_.Draw(0, y + bottom_.Height(),
168  left_.Width(), left_.Height(),
169  0, 0, left_.Width(), left_.Height(), sprite, easy::kCopyRgba);
170  }
171  for (Si32 y = 0; y < client_size.y; y += right_.Height()) {
172  right_.Draw(size.x - right_.Width(), y + bottom_.Height(),
173  right_.Width(), right_.Height(),
174  0, 0, right_.Width(), right_.Height(), sprite, easy::kCopyRgba);
175  }
176  lower_left_.Draw(0, 0,
177  lower_left_.Width(), lower_left_.Height(),
178  0, 0, lower_left_.Width(), lower_left_.Height(), sprite, easy::kCopyRgba);
179  lower_right_.Draw(size.x - lower_right_.Width(), 0,
180  lower_right_.Width(), lower_right_.Height(),
181  0, 0, lower_right_.Width(), lower_right_.Height(), sprite, easy::kCopyRgba);
182  upper_left_.Draw(0, size.y - upper_left_.Height(),
183  upper_left_.Width(), upper_left_.Height(),
184  0, 0, upper_left_.Width(), upper_left_.Height(), sprite, easy::kCopyRgba);
185  upper_right_.Draw(size.x - upper_right_.Width(), size.y - upper_right_.Height(),
186  upper_right_.Width(), upper_right_.Height(),
187  0, 0, upper_right_.Width(), upper_right_.Height(), sprite, easy::kCopyRgba);
188 
189  return sprite;
190  }
